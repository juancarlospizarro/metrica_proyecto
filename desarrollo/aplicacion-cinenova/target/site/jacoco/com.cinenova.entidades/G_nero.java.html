package com.cinenova.entidades;

/**
 * Enumerado que recoge los diferentes géneros de las películas
 * 
 * @author Juan Carlos Pizarro Alonso
 */
public enum Género {

    /**
     * Categoría de género Comedia
     */
    COMEDIA,

    /**
     * Categoría de género Drama
     */
    DRAMA,

    /**
     * Categoría de género Acción
     */
    ACCION,

    /**
     * Categoría de género Ciencia Ficción
     */
    CIENCIA_FICCION,

    /**
     * Categoría de género Animación
     */
    ANIMACION,

    /**
     * Categoría de género Documental
     */
    DOCUMENTAL,
    
    /**
     * Categoría de género Musical
     */
    MUSICAL,
    
    /**
     * Categoría de género Romance
     */
    ROMANCE
}
